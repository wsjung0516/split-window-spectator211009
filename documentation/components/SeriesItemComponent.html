    img {

      object-fit: fill;

      border: blue solid 4px ;
    }
  `],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class SeriesItemComponent implements OnInit, OnChanges, AfterViewInit {

  borderColor: any;
